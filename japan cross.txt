class GridComponent { accept(visitor) { throw new Error("Method not implemented"); } }
class Cell extends GridComponent {
    constructor(row, col) { super(); this.row = row; this.col = col; this.state = 0; }
    accept(visitor) { visitor.visitCell(this); }
    setState(newState) { this.state = newState; }
}
class GridGroup extends GridComponent {
    constructor() { super(); this.children = []; }
    add(component) { this.children.push(component); }
    accept(visitor) { visitor.visitGroup(this); for (let child of this.children) child.accept(visitor); }
}

class Command { execute() {} undo() {} }
class SetCellStateCommand extends Command {
    constructor(cell, newState) { super(); this.cell = cell; this.oldState = cell.state; this.newState = newState; }
    execute() { this.cell.setState(this.newState); }
    undo() { this.cell.setState(this.oldState); }
}
class MacroCommand extends Command {
    constructor() { super(); this.commands = []; }
    add(cmd) { this.commands.push(cmd); return this; }
    execute() { this.commands.forEach(c => c.execute()); }
    undo() { for(let i = this.commands.length - 1; i >= 0; i--) this.commands[i].undo(); }
}

class CommandManager {
    constructor() { this.undoStack = []; this.redoStack = []; }
    pushMacro(macro) { this.undoStack.push(macro); this.redoStack = []; }
    undo() { if (!this.undoStack.length) return null; const cmd = this.undoStack.pop(); cmd.undo(); this.redoStack.push(cmd); return cmd; }
    redo() { if (!this.redoStack.length) return null; const cmd = this.redoStack.pop(); cmd.execute(); this.undoStack.push(cmd); return cmd; }
    clear() { this.undoStack = []; this.redoStack = []; }
}

class Visitor { visitCell(cell) {} visitGroup(group) {} }
class WinCheckVisitor extends Visitor {
    constructor(levelData) { super(); this.levelData = levelData; this.errors = []; this.missing = 0; }
    visitCell(cell) {
        const expected = this.levelData[cell.row][cell.col];
        if (expected === 0 && cell.state === 1) this.errors.push({row: cell.row, col: cell.col});
        else if (expected === 1 && cell.state !== 1) this.missing++;
    }
}
class ExtractStateVisitor extends Visitor {
    constructor() { super(); this.stateMatrix = []; this.currentRow = []; }
    visitCell(cell) { this.currentRow.push(cell.state); }
    visitGroup(group) { if (this.currentRow.length > 0) { this.stateMatrix.push([...this.currentRow]); this.currentRow = []; } }
    getState() { if (this.currentRow.length > 0) this.stateMatrix.push([...this.currentRow]); return this.stateMatrix; }
}

class GameCore {
    constructor() {
        this.rootGroup = new GridGroup(); this.cmdManager = new CommandManager();
        this.currentLevel = null; this.size = 0; this.timeSeconds = 0;
        this.isEditorMode = false; this.isWon = false;
    }
    loadLevel(levelObj, savedState = null, savedTime = 0) {
        this.currentLevel = levelObj; this.size = levelObj.size; this.isEditorMode = false; this.isWon = false;
        this.cmdManager.clear(); this.timeSeconds = savedTime; this.rootGroup = new GridGroup();
        for(let i=0; i<this.size; i++) {
            let row = new GridGroup();
            for(let j=0; j<this.size; j++) {
                let cell = new Cell(i, j);
                if (savedState) cell.setState(savedState[i][j]);
                row.add(cell);
            }
            this.rootGroup.add(row);
        }
    }
    initEditor(size) {
        this.isEditorMode = true; this.size = size; this.isWon = false; this.rootGroup = new GridGroup(); this.cmdManager.clear();
        for(let i=0; i<size; i++) { let row = new GridGroup(); for(let j=0; j<size; j++) row.add(new Cell(i, j)); this.rootGroup.add(row); }
    }
    getCell(r, c) { return this.rootGroup.children[r].children[c]; }
    
    getHintableCells() {
        if(this.isEditorMode) return [];
        let cells = [];
        for(let r=0; r<this.size; r++) {
            for(let c=0; c<this.size; c++) {
                let exp = this.currentLevel.data[r][c], cur = this.getCell(r, c).state;
                if (exp === 1 && cur !== 1) cells.push({r, c, target: 1}); 
                if (exp === 0 && cur === 1) cells.push({r, c, target: 2}); 
            }
        }
        return cells;
    }

    checkWin() {
        if (this.isEditorMode) return null;
        const visitor = new WinCheckVisitor(this.currentLevel.data);
        this.rootGroup.accept(visitor);
        return { errors: visitor.errors, missing: visitor.missing };
    }
    getCurrentState() { const v = new ExtractStateVisitor(); this.rootGroup.accept(v); return v.getState(); }
}

class StorageAdapter {
    saveProgress(levelId, stateMatrix, timeSeconds, hintsLeft) { 
        let all = this.getAllProgress();
        all[levelId] = { stateMatrix, timeSeconds, hintsLeft };
        localStorage.setItem('jcross_all_progress', JSON.stringify(all)); 
    }
    getAllProgress() { const r = localStorage.getItem('jcross_all_progress'); return r ? JSON.parse(r) : {}; }
    loadProgress(levelId) { return this.getAllProgress()[levelId] || null; }
    clearProgress(levelId) { 
        let all = this.getAllProgress();
        delete all[levelId];
        localStorage.setItem('jcross_all_progress', JSON.stringify(all));
    }

    setLastPlayed(levelId) { localStorage.setItem('jcross_last_played', levelId); }
    getLastPlayed() { return localStorage.getItem('jcross_last_played'); }

    saveCustomLevels(levels) { localStorage.setItem('jcross_custom', JSON.stringify(levels)); }
    loadCustomLevels() { const r = localStorage.getItem('jcross_custom'); return r ? JSON.parse(r) : []; }
    
    getBestTimes() { const r = localStorage.getItem('jcross_best_times'); return r ? JSON.parse(r) : {}; }
    getBestTime(levelId) { return this.getBestTimes()[levelId] || null; }
    saveBestTime(levelId, timeSeconds) {
        const times = this.getBestTimes(); times[levelId] = timeSeconds;
        localStorage.setItem('jcross_best_times', JSON.stringify(times));
    }
}

class FileAdapter {
    exportToJSON(levelsArray) {
        const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(levelsArray));
        const a = document.createElement('a'); a.href = dataStr; a.download = "custom_levels.json";
        document.body.appendChild(a); a.click(); a.remove();
    }
    importFromJSON(file, callback) {
        const reader = new FileReader();
        reader.onload = e => { try { callback(JSON.parse(e.target.result)); } catch { alert("Ошибка парсинга JSON"); } };
        reader.readAsText(file);
    }
}

class DOMAdapter {
    constructor(controller) {

        this.container = document.getElementById("gameContainer");

        this.bestTimeDisp = document.getElementById("bestTimeDisplay");
        this.delBtn = document.getElementById("deleteLvlBtn");
        this.hintBtn = document.getElementById("hintBtn");
        this.topCluesDOM = []; this.leftCluesDOM = [];

        document.addEventListener('mouseup', () => this.ctrl.handleGlobalMouseUp());
        document.addEventListener('mouseleave', () => this.ctrl.handleGlobalMouseUp());
        document.addEventListener('dragstart', e => e.preventDefault());
        this.bindEvents();
    }

    bindEvents() {
        document.getElementById("levelSelect").addEventListener("change", e => {
            this.delBtn.disabled = e.target.value.startsWith("lvl_"); 
            this.ctrl.onLevelChange(e.target.value);
        });
        
        document.getElementById("checkBtn").addEventListener("click", () => this.ctrl.onCheckWin());
        document.getElementById("resetBtn").addEventListener("click", () => this.ctrl.onResetLevel());
        document.getElementById("deleteLvlBtn").addEventListener("click", () => this.ctrl.deleteCurrentLevel());
        document.getElementById("saveProgressBtn").addEventListener("click", () => this.ctrl.forceSaveProgress());
        this.hintBtn.addEventListener("click", () => this.ctrl.useHint());

        document.getElementById("toggleEditorBtn").addEventListener("click", () => this.ctrl.toggleEditorMode());
        document.getElementById("exitEditorBtn").addEventListener("click", () => this.ctrl.toggleEditorMode());
        document.getElementById("createBlankBtn").addEventListener("click", () => this.ctrl.initEditorGrid(parseInt(document.getElementById("editorSize").value)));
        document.getElementById("saveCustomLvlBtn").addEventListener("click", () => this.ctrl.saveLevelFromEditor(document.getElementById("customLevelName").value));
        document.getElementById("exportJsonBtn").addEventListener("click", () => this.ctrl.exportJSON());
        document.getElementById("importJsonFile").addEventListener("change", e => { if(e.target.files.length) this.ctrl.importJSON(e.target.files[0]); });
        
        document.addEventListener('keydown', (e) => {
            if (e.target.tagName === 'INPUT') return;
            if (e.ctrlKey && (e.key === 'z' || e.key === 'я')) { e.preventDefault(); this.ctrl.onUndo(); }
            if (e.ctrlKey && (e.key === 'y' || e.key === 'н' || (e.shiftKey && (e.key === 'Z' || e.key === 'Я')))) { e.preventDefault(); this.ctrl.onRedo(); }
        });
    }

    updateHintBtn(hintsLeft, isWon) {
        this.hintBtn.innerText = `💡 Подсказки: ${hintsLeft}`;
        this.hintBtn.disabled = hintsLeft <= 0 || isWon;
    }

    populateLevelSelect(defLvl, custLvl, selId) {
        const sel = document.getElementById("levelSelect"); sel.innerHTML = "";
        defLvl.forEach(l => sel.add(new Option(`Стандарт: ${l.name}`, l.id)));
        custLvl.forEach(l => sel.add(new Option(`Кастом: ${l.name}`, l.id)));
        sel.value = selId;
        this.delBtn.disabled = selId.startsWith("lvl_"); 
    }

    formatTime(sec) { return `${Math.floor(sec/60).toString().padStart(2,'0')}:${(sec%60).toString().padStart(2,'0')}`; }
    renderTimer(sec) { this.timerDisp.innerText = `Время: ${this.formatTime(sec)}`; }
    renderBestTime(sec) { this.bestTimeDisp.innerText = sec === null ? "Лучшее: --:--" : `Лучшее: ${this.formatTime(sec)}`; }
    showMessage(text, color) { this.msg.innerText = text; this.msg.style.color = color; }

    // Функция для определения блоков закрашенных клеток и их позиций
    _getBlocks(line) {
        let blocks = []; let count = 0; let start = -1;
        for(let i=0; i<line.length; i++) {
            if(line[i] === 1) {
                if(count === 0) start = i;
                count++;
            } else if(count > 0) {
                blocks.push({len: count, start: start, end: i-1}); count = 0;
            }
        }
        if(count > 0) blocks.push({len: count, start: start, end: line.length-1});
        return blocks;
    }

    renderGrid(core) {
        this.container.innerHTML = ''; this.topCluesDOM = []; this.leftCluesDOM = [];
        const size = core.size; const grid = document.createElement('div');
        grid.className = 'grid-container';
        grid.style.gridTemplateColumns = core.isEditorMode ? `repeat(${size}, 40px)` : `90px repeat(${size}, 40px)`;
        grid.style.gridTemplateRows = core.isEditorMode ? `repeat(${size}, 40px)` : `70px repeat(${size}, 40px)`;
        grid.addEventListener('contextmenu', e => e.preventDefault());

        if (!core.isEditorMode) {
            grid.appendChild(Object.assign(document.createElement('div'), {className: 'grid-cell corner-cell'}));
            
            // ВЕРХНИЕ ПОДСКАЗКИ
            for(let j=0; j<size; j++) {
                let colData = []; for(let i=0; i<size; i++) colData.push(core.currentLevel.data[i][j]);
                let blocks = this._getBlocks(colData);
                
                const clueDiv = document.createElement('div'); 
                clueDiv.className = 'grid-cell top-clue';
                clueDiv.style.flexDirection = 'column'; 
                clueDiv.style.justifyContent = 'flex-end'; 
                
                let spans = [];
                if (blocks.length === 0) {
                    clueDiv.innerHTML = "<span>0</span>"; 
                    spans.push(clueDiv.firstChild);
                } else {
                    blocks.forEach(b => {
                        let span = document.createElement('span'); 
                        span.innerText = b.len;
                        clueDiv.appendChild(span); 
                        spans.push(span);
                    });
                }
                grid.appendChild(clueDiv); 
                this.topCluesDOM.push({ el: clueDiv, spans: spans, expectedBlocks: blocks });
            }
            
            // ЛЕВЫЕ ПОДСКАЗКИ
            for(let i=0; i<size; i++) {
                let rowData = core.currentLevel.data[i];
                let blocks = this._getBlocks(rowData);
                
                const leftDiv = document.createElement('div'); 
                leftDiv.className = 'grid-cell left-clue';
                
                let spans = [];
                if (blocks.length === 0) {
                    leftDiv.innerHTML = "<span>0</span>"; 
                    spans.push(leftDiv.firstChild);
                } else {
                    blocks.forEach((b, idx) => {
                        let span = document.createElement('span'); 
                        span.innerText = b.len;
                        if(idx < blocks.length - 1) span.style.marginRight = '6px'; 
                        leftDiv.appendChild(span); 
                        spans.push(span);
                    });
                }
                grid.appendChild(leftDiv); 
                this.leftCluesDOM.push({ el: leftDiv, spans: spans, expectedBlocks: blocks });
                
                this.renderRow(grid, core, i);
            }
        } else { 
            for(let i=0; i<size; i++) this.renderRow(grid, core, i); 
        }
        this.container.appendChild(grid); 
        this.updateClues(core);
    }

    renderRow(gridDOM, core, r) {
        for(let c=0; c<core.size; c++) {
            const cellDom = document.createElement('div');
            cellDom.dataset.r = r; cellDom.dataset.c = c; gridDOM.appendChild(cellDom);
            cellDom.addEventListener('mousedown', e => { e.preventDefault(); this.clearErrors(); this.ctrl.handleMouseDown(r, c, e.button); });
            cellDom.addEventListener('mouseenter', e => this.ctrl.handleDragToCell(r, c));
            this.updateCellDOM(r, c, core.getCell(r, c).state, core.isEditorMode, cellDom);
        }
    }

    updateCellDOM(r, c, state, isEditor, element = null) {
        let el = element || this.container.querySelector(`[data-r="${r}"][data-c="${c}"]`);
        if(!el) return;
        el.className = 'grid-cell play-cell'; el.innerText = '';
        if (state === 1) el.classList.add(isEditor ? 'cell-editor-draw' : 'cell-black');
        else if (state === 2 && !isEditor) { el.classList.add('cell-crossed'); el.innerText = '✕'; }
    }

    fullUpdateCells(core) { for(let r=0; r<core.size; r++) for(let c=0; c<core.size; c++) this.updateCellDOM(r, c, core.getCell(r, c).state, core.isEditorMode); }
    
    // Новая логика: независимое зачеркивание каждой цифры-подсказки
    updateClues(core) {
        if(core.isEditorMode) return;
        let state = core.getCurrentState();

        // Проверка столбцов (Верхние подсказки)
        for(let j=0; j<core.size; j++) {
            let clueData = this.topCluesDOM[j];
            let colState = []; for(let i=0; i<core.size; i++) colState.push(state[i][j]);
            
            if (clueData.expectedBlocks.length === 0) {
                let hasErrors = colState.includes(1);
                clueData.spans[0].classList.toggle('clue-done', !hasErrors);
            } else {
                clueData.expectedBlocks.forEach((block, idx) => {
                    let isDone = true;
                    // Проверяем, закрашены ли все нужные клетки для данного блока
                    for(let i=block.start; i<=block.end; i++) { if(colState[i] !== 1) isDone = false; }
                    // Убеждаемся, что границы не сливаются с соседними нарисованными клетками
                    if (block.start > 0 && colState[block.start - 1] === 1) isDone = false;
                    if (block.end < core.size - 1 && colState[block.end + 1] === 1) isDone = false;
                    
                    clueData.spans[idx].classList.toggle('clue-done', isDone);
                });
            }
        }
        
        // Проверка строк (Левые подсказки)
        for(let i=0; i<core.size; i++) {
            let clueData = this.leftCluesDOM[i];
            let rowState = state[i];
            
            if (clueData.expectedBlocks.length === 0) {
                let hasErrors = rowState.includes(1);
                clueData.spans[0].classList.toggle('clue-done', !hasErrors);
            } else {
                clueData.expectedBlocks.forEach((block, idx) => {
                    let isDone = true;
                    for(let c=block.start; c<=block.end; c++) { if(rowState[c] !== 1) isDone = false; }
                    if (block.start > 0 && rowState[block.start - 1] === 1) isDone = false;
                    if (block.end < core.size - 1 && rowState[block.end + 1] === 1) isDone = false;
                    
                    clueData.spans[idx].classList.toggle('clue-done', isDone);
                });
            }
        }
    }

    clearErrors() { 
        document.querySelectorAll('.cell-error').forEach(el => el.classList.remove('cell-error')); 
        document.querySelectorAll('.cell-hinted').forEach(el => el.classList.remove('cell-hinted')); 
        this.msg.innerText = ""; 
    }
    showErrors(arr) { arr.forEach(err => { let el = this.container.querySelector(`[data-r="${err.row}"][data-c="${err.col}"]`); if(el) el.classList.add('cell-error'); }); }
    highlightHint(r, c) { let el = this.container.querySelector(`[data-r="${r}"][data-c="${c}"]`); if(el) { el.classList.add('cell-hinted'); setTimeout(() => el.classList.remove('cell-hinted'), 1500); } }

    toggleEditorUI(isEd) {
        document.getElementById('playControls').style.display = isEd ? 'none' : 'flex';
        document.getElementById('bottomControls').style.display = isEd ? 'none' : 'flex';
        document.getElementById('footerPanel').style.display = isEd ? 'none' : 'flex';
        document.getElementById('editorControls').style.display = isEd ? 'flex' : 'none';
        this.msg.innerText = isEd ? "Редактор: Левая кнопка - рисовать, Правая кнопка - ластик" : "";
    }
}

        this.isDragging = false; this.dragTargetState = null; this.currentMacro = null;
        this.initGame();
    }

    getAllLevels() { return [...this.defaultLevels, ...this.customLevels]; }
    getLevelById(id) { return this.getAllLevels().find(l => l.id === id); }

    initGame() {
        let lastPlayed = this.storage.getLastPlayed();
        let startId = this.getLevelById(lastPlayed) ? lastPlayed : this.defaultLevels[0].id;
        
        this.ui.populateLevelSelect(this.defaultLevels, this.customLevels, startId);
        this.loadLevelCore(startId);
    }

    loadLevelCore(id) {
        this.storage.setLastPlayed(id);
        const savedData = this.storage.loadProgress(id);
        
        let state = savedData ? savedData.stateMatrix : null;
        let time = savedData ? savedData.timeSeconds : 0;
        this.hintsLeft = savedData && savedData.hintsLeft !== undefined ? savedData.hintsLeft : 3;

        this.core.loadLevel(this.getLevelById(id), state, time);
        
        this.ui.renderGrid(this.core); 
        this.ui.updateHintBtn(this.hintsLeft, false);
        this.ui.showMessage("", "");
        
        const bestTime = this.storage.getBestTime(id);
        this.ui.renderBestTime(bestTime);
        this.startTimer();
    }

    onLevelChange(id) { 
        if(!this.core.isWon && !this.core.isEditorMode) this.autoSave(); 
        this.loadLevelCore(id); 
    }
    
    onResetLevel() { 
        const msg = this.core.isWon 
            ? "Вы уже решили этот кроссворд! Точно хотите очистить поле и пройти его заново?" 
            : "Очистить поле и начать заново?";

        if(confirm(msg)) {
            this.storage.clearProgress(this.core.currentLevel.id); 
            this.loadLevelCore(this.core.currentLevel.id); 
        }
    }

    forceSaveProgress() {
        if (this.core.isWon || this.core.isEditorMode) return;
        this.autoSave();
        this.ui.showMessage("Прогресс успешно сохранен!", "#0288d1");
        setTimeout(() => { if(this.ui.msg.innerText === "Прогресс успешно сохранен!") this.ui.msg.innerText = ""; }, 2000);
    }

    autoSave() { this.storage.saveProgress(this.core.currentLevel.id, this.core.getCurrentState(), this.core.timeSeconds, this.hintsLeft); }

    useHint() {
        if (this.hintsLeft <= 0 || this.core.isWon || this.core.isEditorMode) return;
        const hintable = this.core.getHintableCells();
        if (hintable.length === 0) return; 

        const target = hintable[Math.floor(Math.random() * hintable.length)];
        const cell = this.core.getCell(target.r, target.c);
        const cmd = new SetCellStateCommand(cell, target.target);
        cmd.execute();
        this.core.cmdManager.pushMacro(new MacroCommand().add(cmd));

        this.hintsLeft--;
        this.ui.updateHintBtn(this.hintsLeft, this.core.isWon);
        this.ui.highlightHint(target.r, target.c);
        this.afterUndoRedo();
    }

    deleteCurrentLevel() {
        const id = document.getElementById("levelSelect").value;
        if (id.startsWith("lvl_")) return; 
        if (!confirm("Вы уверены, что хотите удалить этот уровень?")) return;
        this.customLevels = this.customLevels.filter(l => l.id !== id);
        this.storage.saveCustomLevels(this.customLevels);
        this.storage.clearProgress(id);
        this.initGame();
    }

    handleMouseDown(r, c, btn) {
        if (this.core.isWon) return;
        this.isDragging = true; this.currentMacro = new MacroCommand();
        const cell = this.core.getCell(r, c);
        
        if (this.core.isEditorMode) {
            if (btn === 0) this.dragTargetState = (cell.state === 1) ? 0 : 1;
            else if (btn === 2) this.dragTargetState = 0;
            else this.dragTargetState = cell.state;
        } else {
            if (btn === 0) this.dragTargetState = (cell.state === 1) ? 0 : 1;
            else if (btn === 2) this.dragTargetState = (cell.state === 2) ? 0 : 2;
            else this.dragTargetState = cell.state;
        }
        this.handleDragToCell(r, c);
    }

    handleDragToCell(r, c) {
        if (!this.isDragging || this.core.isWon) return;
        const cell = this.core.getCell(r, c);
        if (cell.state !== this.dragTargetState) {
            const cmd = new SetCellStateCommand(cell, this.dragTargetState);
            cmd.execute(); this.currentMacro.add(cmd);
            this.ui.updateCellDOM(r, c, cell.state, this.core.isEditorMode);
            this.ui.updateClues(this.core);
        }
    }

    handleGlobalMouseUp() {
        if (this.isDragging) {
            this.isDragging = false;
            if (this.currentMacro && this.currentMacro.commands.length > 0) {
                this.core.cmdManager.pushMacro(this.currentMacro);
                this.afterUndoRedo(); 
            }
            this.currentMacro = null;
        }
    }

    onUndo() { if(!this.core.isWon) { this.core.cmdManager.undo(); this.afterUndoRedo(); } }
    onRedo() { if(!this.core.isWon) { this.core.cmdManager.redo(); this.afterUndoRedo(); } }

    afterUndoRedo() {
        this.ui.clearErrors(); this.ui.fullUpdateCells(this.core); this.ui.updateClues(this.core);
        if(!this.core.isEditorMode && !this.core.isWon) {
            this.autoSave();
            const check = this.core.checkWin();
            if (check.missing === 0 && check.errors.length === 0) this.onCheckWin();
        }
    }

    onCheckWin() {
        const result = this.core.checkWin();
        if (result.errors.length > 0) {
            this.ui.showErrors(result.errors); this.ui.showMessage(`Ошибок: ${result.errors.length}. Исправьте подсвеченные клетки.`, "#c62828");
        } else if (result.missing > 0) {
            this.ui.showMessage("Ошибок нет, но кроссворд ещё не закончен!", "#1e3c72");
        } else {
            this.core.isWon = true; 
            this.stopTimer(); 
            this.storage.clearProgress(this.core.currentLevel.id); 
            this.ui.updateHintBtn(this.hintsLeft, true);

            const oldBest = this.storage.getBestTime(this.core.currentLevel.id);
            if (oldBest === null || this.core.timeSeconds < oldBest) {
                this.storage.saveBestTime(this.core.currentLevel.id, this.core.timeSeconds);
                this.ui.renderBestTime(this.core.timeSeconds);
                this.ui.showMessage(`Победа! НОВЫЙ РЕКОРД: ${this.ui.formatTime(this.core.timeSeconds)} 🎉`, "#2e7d32");
            } else {
                this.ui.showMessage(`Победа! Ваше время: ${this.ui.formatTime(this.core.timeSeconds)}.`, "#2e7d32");
            }
        }
    }

    startTimer() {
        this.stopTimer(); this.ui.renderTimer(this.core.timeSeconds);
        this.timerId = setInterval(() => {
            if(!this.core.isWon) {
                this.core.timeSeconds++; this.ui.renderTimer(this.core.timeSeconds);
                if(this.core.timeSeconds % 5 === 0) this.autoSave();
            }
        }, 1000);
    }
    stopTimer() { if (this.timerId) clearInterval(this.timerId); }

    toggleEditorMode() {
        if (!this.core.isEditorMode) { 
            this.autoSave();
            this.stopTimer(); this.ui.toggleEditorUI(true); this.initEditorGrid(5); 
        } else { 
            this.ui.toggleEditorUI(false); 
            this.initGame(); 
        }
    }
    
    initEditorGrid(size) { this.core.initEditor(size); this.ui.renderGrid(this.core); }
    
    saveLevelFromEditor(name) {
        if(!name.trim()) return alert("Введите название уровня!");
        const cleanData = this.core.getCurrentState().map(r => r.map(c => c === 1 ? 1 : 0));
        
        const newId = "custom_" + Date.now();
        this.customLevels.push({ id: newId, name: name.trim(), size: this.core.size, data: cleanData });
        this.storage.saveCustomLevels(this.customLevels);
        
        alert("Уровень сохранен!"); 
        
        this.storage.setLastPlayed(newId);
        this.toggleEditorMode();
    }
    
    exportJSON() { if(!this.customLevels.length) return alert("У вас нет кастомных уровней."); this.fileManager.exportToJSON(this.customLevels); }
    
    importJSON(file) {
        this.fileManager.importFromJSON(file, importedLevels => {
            if(!Array.isArray(importedLevels)) return alert("Ошибка: Файл должен содержать массив уровней.");
            const valid = [];
            for(let l of importedLevels) {
                if(!l.id || !l.name || typeof l.size !== 'number' || !Array.isArray(l.data) || l.data.length !== l.size) continue;
                let isMatrixValid = true;
                for(let row of l.data) {
                    if(!Array.isArray(row) || row.length !== l.size) { isMatrixValid = false; break; }
                    for(let cell of row) if(cell !== 0 && cell !== 1) { isMatrixValid = false; break; }
                }
                if(isMatrixValid) valid.push(l);
            }
            if(valid.length === 0) return alert("В файле нет корректных уровней!");
            this.customLevels = [...this.customLevels, ...valid];
            this.storage.saveCustomLevels(this.customLevels);
            this.ui.populateLevelSelect(this.defaultLevels, this.customLevels, this.core.currentLevel.id);
            alert(`Успешно импортировано уровней: ${valid.length}`);
        });
    }
}

window.onload = () => window.gameApp = new GameController();
